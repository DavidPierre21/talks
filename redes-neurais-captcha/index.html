  inputs = layers.Input((40, 40, 3))

  x = layers.Conv2D(32, 9, activation='relu')(x)
  x = layers.MaxPool2D((2, 2))(x)
  x = layers.Dropout(0.25)(x)
  x = layers.Flatten()(x)
  x = layers.Dense(640)(x)
  x = layers.Dropout(0.5)(x)
